[docs]def FAIR(item=None, f=None, T=None, h=None, Pr=None, phi=None):
    list_variables = [item, f, T, h, Pr, phi]
    nargin = sum(x is not None for x in list_variables)

    #  Derived from equations
    #  Convertion factors
    R2K = 0.5556
    BTUlbm2Jkg = 2326
    psia2pa = 6895
    BTUlbR2JKgK = 4184
    fts2ms = 0.3048

    nargin = 2 + nargin
    if f > 0.0676:
        print('f cannot be greater than 0.0676')

    if item == 1:  # T is known
        if nargin > 2:
            T = T/R2K
            h, Pr, phi, Cp, R, gamma, a = unFAIR(T, f)
        else:
            print('T must be defined for case 1')

    elif item == 2:  # h is known
        if nargin > 3:
            h = h/BTUlbm2Jkg
            x0 = np.array([166.667, 2222.222])
            T = optimize.fminbound(lambda T: np.abs(h-find_h(f, T)), x0)
            h, Pr, phi, Cp, R, gamma, a = unFAIR(T, f)
        else:
            print('h must be defined for case 2')

    elif item == 3:  # Pr is known
        if nargin > 4:
            Pr = Pr/psia2pa
            x0 = np.array([166.667, 2222.222])
            T = optimize.fminbound(lambda T: np.abs(Pr-find_Pr(f, T)), x0)
            h, Pr, phi, Cp, R, gamma, a = unFAIR(T, f)
        else:
            print('Pr must be defined for case 2')

    elif item == 4:  # phi is known
        if nargin > 5:
            phi = phi/BTUlbR2JKgK
            x0 = np.array([166.667, 2222.222])
            T = optimize.fminbound(lambda T: np.abs(phi-find_phi(f, T)), x0)
            h, Pr, phi, Cp, R, gamma, a = unFAIR(T, f)
        else:
            print(' must be defined for case 2')

    T = T*R2K
    h = h*BTUlbm2Jkg
    Pr = Pr*psia2pa
    phi = phi*BTUlbR2JKgK
    Cp = Cp*BTUlbR2JKgK
    R = R*BTUlbR2JKgK
    a = a*fts2ms

    return T, h, Pr, phi, Cp, R, gamma, a

[docs]def unFAIR(T, FAR):

    BTU_lbm_to_ft2_s2 = 25037.00

    [Cp_a, h_a, phi_a] = AFPROP_A(T)
    [Cp_p, h_p, phi_p] = AFPROP_P(T)

    # ============ Equation 4.26 a, b, c, d ===================
    R = 1.9857117/(28.97-FAR*0.946186)  # BTU ./( lbm R)
    Cp = (Cp_a+FAR*Cp_p)/(1+FAR)
    h = (h_a+FAR*h_p)/(1+FAR)
    phi = (phi_a+FAR*phi_p)/(1+FAR)
    # ============ Equation 2.55 - " reduced pressure " =======
    phi_ref = 1.578420959  # BTU ./( lbm R) phi@492 .00 R

    Pr = np.exp((phi-phi_ref)/R)

    gamma = Cp/(Cp-R)
    a = np.sqrt(gamma*R*BTU_lbm_to_ft2_s2*T)

    return h, Pr, phi, Cp, R, gamma, a

    # ===== Define coeficients from Table 2.2 for air alone ======
    A0 = 2.5020051E-01

    A6 = -1.4526770E-21
    A7 = 1.0115540E-25
    h_ref = -1.7558886  # BTU ./lbm

    # ====== Equations 2.60 , 2.61 , 2.62 for air alone ===========
    Cp_a, h_a, phi_a = AFPROP(
        T, A0, A1, A2, A3, A4, A5, A6, A7, h_ref, phi_ref
    )

[docs]def AFPROP(T, A0, A1, A2, A3, A4, A5, A6, A7, h_ref, phi_ref):
    Cp = (A0
          + A1*T
          + A2*T**2
          + A3*T**3
          + A4*T**4
          + A5*T**5
          + A6*T**6
          + A7*T**7)

    h = (h_ref
         + A0*T
         + (A1/2)*T**2
         + (A2/3)*T**3
         + (A3/4)*T**4
         + (A4/5)*T**5
         + (A5/6)*T**6
         + (A6/7)*T**7
         + (A7/8)*T**8)

    phi = (phi_ref
           + A0*np.log(T)
           + A1*T
           + A2/2*T**2
           + A3/3*T**3
           + A4/4*T**4
           + A5/5*T**5
           + A6/6*T**6
           + A7/7*T**7)
    return Cp, h, phi
